\]

\begin{itemize}
\item Where:
\begin{itemize}
\item $H$ represents the height of the image
\item $W$ represents the width of the image
\item $C$ denotes the number of image channels
\end{itemize}

\item Since most medical X-ray images are grayscale, the number of channels is typically:
\end{itemize}

\[
C = 1
\]

\vspace{0.3cm}

\textbf{Feature Extraction Using Convolutional Neural Networks}

\begin{itemize}
\item The first stage of the learning model extracts meaningful visual patterns from the input images using convolutional layers.
\end{itemize}

\[
F = f_{conv}(X, W_c) + b
\]

\begin{itemize}
\item Where:
\begin{itemize}
\item $F$ denotes the extracted feature map
\item $W_c$ represents the convolution kernel weights
\item $b$ denotes the bias parameter
\item $f_{conv}$ indicates the convolution operation applied to the input image
\end{itemize}

\item The resulting feature map can be expressed as:
\end{itemize}

\[
F \in \mathbb{R}^{H' \times W' \times D}
\]

\begin{itemize}
\item Where:
\begin{itemize}
\item $H'$ and $W'$ denote the spatial dimensions after convolution
\item $D$ represents the number of extracted feature channels
\end{itemize}
\end{itemize}

\vspace{0.3cm}

\textbf{Channel Attention Mechanism}

\begin{itemize}
\item Channel attention is introduced to identify which feature channels carry the most useful diagnostic information.
\end{itemize}

\[
M_c(F) = \sigma(MLP(AvgPool(F)) + MLP(MaxPool(F)))
\]

\begin{itemize}
\item Where:
\begin{itemize}
\item $M_c$ represents the channel attention map
\item $\sigma$ denotes the sigmoid activation function
\item $MLP$ represents a multi-layer perceptron
\end{itemize}

\item The refined feature representation after channel attention can be computed as:
\end{itemize}

\[
F' = M_c(F) \otimes F
\]

\begin{itemize}
\item Here $\otimes$ denotes element-wise multiplication between the attention map and the feature tensor.
\end{itemize}

\vspace{0.3cm}

\textbf{Spatial Attention Mechanism}

\begin{itemize}
\item Spatial attention is used to determine which regions of the image are more important for diagnosis.
\end{itemize}

\[
M_s(F') = \sigma(f^{7 \times 7}([AvgPool(F'); MaxPool(F')]))
\]

\begin{itemize}
\item Where:
\begin{itemize}
\item $M_s$ denotes the spatial attention map
\item $f^{7 \times 7}$ represents convolution using a $7\times7$ kernel
\end{itemize}

\item The updated feature representation is given by:
\end{itemize}

\[
F'' = M_s(F') \otimes F'
\]

\vspace{0.3cm}

\textbf{Adaptive Attention Feature Fusion}

\begin{itemize}
\item To effectively integrate information from different network layers, an adaptive fusion strategy is applied.
\end{itemize}

\[
F_{fusion} = \sum_{i=1}^{n} \alpha_i F_i
\]

\begin{itemize}
\item Where:
\begin{itemize}
\item $F_i$ represents features extracted from the $i^{th}$ layer
\item $\alpha_i$ represents the adaptive weight assigned to each feature map
\end{itemize}

\item The weights are constrained such that their total contribution remains balanced:
\end{itemize}

\[
\sum_{i=1}^{n} \alpha_i = 1
\]

\vspace{0.3cm}

\textbf{Classification Layer}

\begin{itemize}
\item The fused features are then passed through a classification layer to determine the final disease category.
\end{itemize}

\[
P(y|x) = \frac{e^{z_i}}{\sum_{j=1}^{K} e^{z_j}}
\]

\begin{itemize}
\item Where:
\begin{itemize}
\item $K$ denotes the number of possible disease classes
\item $z_i$ represents the output of the final neural layer
\end{itemize}

\item The predicted class is determined using:
\end{itemize}

\[
\hat{y} = \arg\max_i P(y_i|x)
\]

\vspace{0.3cm}

\textbf{Loss Function}

\begin{itemize}
\item During training, the model parameters are optimized using the cross-entropy loss function.
\end{itemize}

\[
L = - \sum_{i=1}^{N} y_i \log(\hat{y_i})
\]

\begin{itemize}
\item Where:
\begin{itemize}
\item $N$ represents the total number of training samples
\item $y_i$ denotes the true class label
\item $\hat{y_i}$ represents the predicted probability
\end{itemize}
\end{itemize}

\vspace{0.3cm}

\textbf{Model Optimization}

\begin{itemize}
\item Model parameters are updated iteratively using gradient descent.
\end{itemize}

\[
W_{t+1} = W_t - \eta \nabla L(W_t)
\]

\begin{itemize}
\item Where:
\begin{itemize}
\item $\eta$ denotes the learning rate
\item $\nabla L$ represents the gradient of the loss function
\end{itemize}
\end{itemize}

\vspace{0.3cm}

\textbf{Performance Evaluation Metrics}

\begin{itemize}
\item The effectiveness of the proposed model is assessed using several standard evaluation metrics.
\end{itemize}

\textbf{Accuracy}

\[
Accuracy = \frac{TP + TN}{TP + TN + FP + FN}
\]

\textbf{Precision}

\[
Precision = \frac{TP}{TP + FP}
\]

\textbf{Recall}

\[
Recall = \frac{TP}{TP + FN}
\]

\textbf{F1 Score}

\[
F1 = 2 \times \frac{Precision \times Recall}{Precision + Recall}
\]
